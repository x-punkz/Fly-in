### Tela responsiva e janela virtual
```python
# gui.py
self.virtual_window = pygame.Surface((self.base_width, self.base_height), pygame.SRCALPHA)
self.game = pygame.Surface((int(self.base_width * 3 / 4), self.base_height), pygame.SRCALPHA)
self.menu = pygame.Surface((int(self.base_width * 1 / 4), self.base_height), pygame.SRCALPHA)

self.window = pygame.display.set_mode((self.width, self.height), pygame.RESIZABLE)
```

```python
# gui.py
def render_to_window(self) -> None:
    scale_x, scale_y, _, _ = self.get_window_scale()
    scaled_surface = pygame.transform.smoothscale(
        self.virtual_window,
        (int(self.base_width * scale_x), int(self.base_height * scale_y))
    )
    self.window.blit(scaled_surface, (0, 0))
```
